import ReactMarkdown from "react-markdown";
import remarkMath from "remark-math";
import rehypeKatex from "rehype-katex";
import "katex/dist/katex.min.css"; // 引入公式样式

interface Props {
  role: "user" | "ai";
  content: string;
}

export default function ChatMessage({ role, content }: Props) {
  const isUser = role === "user";

  return (
    <div className={`flex w-full ${isUser ? "justify-end" : "justify-start"} mb-4`}>
      <div
        className={`max-w-[80%] p-4 rounded-lg overflow-x-auto ${
          isUser 
            ? "bg-blue-600 text-white rounded-br-none" 
            : "bg-gray-100 text-gray-800 rounded-bl-none"
        }`}
      >
        {/* ✅ 修复核心：
           1. 创建一个 div 作为 markdown 的容器
           2. 把原来的 className="prose ..." 移到这个 div 上
        */}
        <div className="prose prose-sm dark:prose-invert break-words max-w-none">
            <ReactMarkdown
              remarkPlugins={[remarkMath]}
              rehypePlugins={[rehypeKatex]}
            >
              {content}
            </ReactMarkdown>
        </div>
      </div>
    </div>
  );
}
