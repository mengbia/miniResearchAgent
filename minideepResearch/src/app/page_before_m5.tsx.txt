"use client";

import { useChatStore } from "@/store/useChatStore";
import ChatMessage from "@/components/Chat/ChatMessage";
import { nanoid } from "nanoid";
import { useRef, useEffect } from "react";

export default function Home() {
  const {
    messages, input, isLoading,
    setInput, addMessage, setLoading, updateLastMessage,
    addStepToLastMessage, completeLastStep // 引入新 Action
  } = useChatStore();

  const messagesEndRef = useRef<HTMLDivElement>(null);

  // 自动滚动
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages, messages.flatMap(m => m.steps)]); // 当消息或步骤变化时都滚动

  const handleSend = async () => {
    if (!input.trim() || isLoading) return;

    const userMsg = input;
    setInput("");

    // 1. 添加用户消息
    addMessage({ role: "user", content: userMsg, id: nanoid() });
    setLoading(true);

    // 2. 预占位 AI 消息
    addMessage({ role: "ai", content: "", id: nanoid() });

    try {
      const response = await fetch("/api/chat", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ message: userMsg }),
      });

      if (!response.ok) throw new Error(response.statusText);

      const reader = response.body?.getReader();
      const decoder = new TextDecoder();

      if (!reader) return;

      let buffer = ""; // 缓冲区，用于处理被切断的 JSON

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        // 解码当前块
        const chunk = decoder.decode(value, { stream: true });
        buffer += chunk;

        // SSE 格式通常以 \n\n 分隔每条消息
        const lines = buffer.split("\n\n");
        // 数组最后一个元素可能是不完整的，留回缓冲区
        buffer = lines.pop() || "";

        for (const line of lines) {
          if (line.startsWith("data: ")) {
            try {
              const jsonStr = line.slice(6); // 去掉 "data: " 前缀
              if (!jsonStr.trim()) continue;

              const data = JSON.parse(jsonStr);

              // === 🌟 核心调度逻辑 🌟 ===

              // 场景 A: 收到文本片段
              if (data.type === "text") {
                updateLastMessage(data.content);
              }
              // 场景 B: 工具开始 (显示"正在搜索...")
              else if (data.type === "tool_start") {
                addStepToLastMessage({
                    id: nanoid(),
                    type: "search",
                    content: `正在搜索: ${data.content.input?.query || "相关信息"}...`,
                    status: "pending"
                });
              }
              // 场景 C: 工具结束 (打勾)
              else if (data.type === "tool_end") {
                completeLastStep();
              }

            } catch (e) {
              console.error("解析错误:", e);
            }
          }
        }
      }

    } catch (error) {
      console.error("请求失败:", error);
      updateLastMessage("\n\n❌ 网络请求错误，请检查 API Key 或网络连接。");
    } finally {
      setLoading(false);
    }
  };

  return (
    <main className="flex flex-col h-screen bg-gray-50 dark:bg-black text-gray-900 dark:text-gray-100">
      {/* 头部 */}
      <header className="flex items-center justify-center py-4 border-b dark:border-gray-800 bg-white dark:bg-gray-900 shadow-sm z-10">
        <h1 className="text-lg font-bold">Mini DeepResearch</h1>
      </header>

      {/* 聊天区域 */}
      <div className="flex-1 overflow-y-auto p-4 md:p-6 scrollbar-thin scrollbar-thumb-gray-300 dark:scrollbar-thumb-gray-700">
        <div className="max-w-3xl mx-auto space-y-6">
          {messages.length === 0 && (
            <div className="text-center mt-32 space-y-4">
              <h2 className="text-2xl font-bold text-gray-700 dark:text-gray-200">
                你想研究什么？
              </h2>
              <p className="text-gray-500">
                支持联网深度搜索、即时数据分析与整合。
              </p>
            </div>
          )}

          {messages.map((msg) => (
            <ChatMessage key={msg.id} message={msg} />
          ))}

          <div ref={messagesEndRef} className="h-4" />
        </div>
      </div>

      {/* 输入区域 */}
      <div className="p-4 bg-white dark:bg-gray-900 border-t dark:border-gray-800">
        <div className="max-w-3xl mx-auto flex gap-3">
          <input
            type="text"
            className="flex-1 p-3.5 bg-gray-100 dark:bg-gray-800 border-0 rounded-xl focus:ring-2 focus:ring-blue-500 focus:outline-none transition-all"
            placeholder="输入问题，例如：分析一下英伟达最新的财报..."
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={(e) => e.key === "Enter" && handleSend()}
            disabled={isLoading}
          />
          <button
            onClick={handleSend}
            disabled={isLoading || !input.trim()}
            className="px-6 py-3.5 bg-blue-600 text-white rounded-xl hover:bg-blue-700 disabled:opacity-50 disabled:cursor-not-allowed font-medium transition-colors shadow-lg shadow-blue-600/20"
          >
            {isLoading ? "研究中..." : "发送"}
          </button>
        </div>
        <p className="text-center text-xs text-gray-400 mt-3">
          Powered by LangGraph & Tavily
        </p>
      </div>
    </main>
  );
}
